# Merges all enabled scalarized lakes into unified_master.jsonl.

import json, math
from pathlib import Path

ROOT       = Path(__file__).resolve().parent.parent
CONFIG     = ROOT / "configs" / "volumes.json"
UNIFIED    = ROOT / "lakes" / "unified"

def find_scalarized(name):
    for p in [UNIFIED / f"{name}_scalarized.jsonl",
              UNIFIED / f"{name}.jsonl"]:
        if p.exists(): return p
    return None

def main():
    with CONFIG.open("r", encoding="utf-8") as f:
        volumes = json.load(f)["volumes"]

    enabled  = [(n,c) for n,c in volumes.items() if c.get("enabled")]
    disabled = [(n,c) for n,c in volumes.items() if not c.get("enabled")]
    print(f"Volumes: {len(volumes)} total, "
          f"{len(enabled)} enabled, {len(disabled)} disabled")

    domain_stats = {}
    total = 0

    with MASTER.open("w", encoding="utf-8") as fout:
        for lake_name, cfg in volumes.items():
            if not cfg.get("enabled"): continue
            path = find_scalarized(lake_name)
            if not path: continue
            domain = cfg.get("domain", lake_name)
            n = 0

            with path.open("r", encoding="utf-8") as fin:
                for line in fin:
                    line = line.strip()
                    if not line: continue
                    rec = json.loads(line)
                    scalar = float(rec.get("scalar_klc") or
                                   rec.get("scalar_kls") or 0.0)
                    if not math.isfinite(scalar): scalar = 0.0
                    rec["domain"] = domain
                    fout.write(json.dumps(rec, ensure_ascii=False)+"\n")

                    if domain not in domain_stats:
                        domain_stats[domain] = {
                            "n":0,"nonzero":0,"sum":0.0,
                            "sum_sq":0.0,"min":1e18,"max":-1e18}
                    ds = domain_stats[domain]
                    ds["n"] += 1
                    if scalar != 0.0:
                        ds["nonzero"] += 1
                        ds["sum"] += scalar
                        ds["sum_sq"] += scalar*scalar
                        ds["min"] = min(ds["min"], scalar)
                        ds["max"] = max(ds["max"], scalar)
                    n += 1
                    total += 1

            nz = domain_stats.get(domain,{}).get("nonzero",0)
            print(f"  [{lake_name}] domain={domain}  "
                  f"n={n:,}  nonzero={nz:,}")

    print(f"\nTotal records written: {total:,}")
    print(f"Unified master: {MASTER}")

if __name__ == "__main__":
    main()

# build_chaos_nulls.py — key section
# For each domain: generate chaos null (uniform random over
# domain scalar range) and scramble null (shuffled real scalars).
# Compute per-domain chaos z-scores.
# Full script at: github.com/KishLattice16pi/LatticeLab
# AUTHORS: Timothy John Kish & Mondy

import json, math, random
from pathlib import Path

PI    = math.pi
K_GEO = 16.0 / PI
MODULI = [15/PI, 16/PI, 17/PI]   # candidate harmonic moduli

BINS = 24  # harmonic container

def compute_lock_rate(scalars, modulus):
    """Fraction of scalars within 5% of a modulus node."""
    if not scalars: return 0.0
    locked = sum(1 for s in scalars
                 if any(abs((s % modulus) - k*modulus) < 0.05*modulus
                        for k in range(BINS+1)))
    return locked / len(scalars)

def chaos_z_score(real_scalars, n_chaos=10000):
    """
    Z-score: how many sigma above chaos null is the real lock rate.
    Chaos null: uniform random over same scalar range.
    """
    lo, hi = min(real_scalars), max(real_scalars)
    results = {}
    for mod in MODULI:
        real_rate = compute_lock_rate(real_scalars, mod)
        chaos_rates = [
            compute_lock_rate(
                [random.uniform(lo, hi) for _ in real_scalars], mod)
            for _ in range(50)
        ]
        mu_c = sum(chaos_rates)/len(chaos_rates)
        sd_c = (sum((r-mu_c)**2 for r in chaos_rates)/len(chaos_rates))**0.5
        z = (real_rate - mu_c) / max(sd_c, 1e-10)
        results[mod] = {"real": real_rate, "chaos": mu_c,
                        "delta": real_rate - mu_c, "z": z}
    return results
